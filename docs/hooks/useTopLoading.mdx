# useTopLoading

:::tip

当前组件需要的 frame 顶部加载 loading 效果

:::

## 代码

```jsx
import { Loading } from '@shopify/polaris'
import { useCallback, useMemo, useState } from 'react'

export const useTopLoading = () => {
  const [isLoading, setIsLoading] = useState(false)

  const hiddenTopLoading = useCallback(() => setIsLoading(false), [])
  const showTopLoading = useCallback(() => setIsLoading(true), [])

  const TopLoadingJSX = useMemo(() => (isLoading ? <Loading /> : null), [isLoading])

  return {
    TopLoadingJSX,
    hiddenTopLoading,
    showTopLoading,
  }
}
```
